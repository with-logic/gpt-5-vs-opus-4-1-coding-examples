        02 / One call, three languages
        [ DX=MAX ]
      

      
Copy. Paste. Ship.


      

        
        

          

            
Send events

            
Fire structured events from any runtime in one HTTP call.

          

          

            
            
            
          

          

            
const res = await fetch("https://api.nullstack.dev/v1/events", {
  method: "POST",
  headers: { "Authorization": `Bearer ${KEY}` },
  body: JSON.stringify({ name: "deploy.start" })
});

            
import requests

res = requests.post(
    "https://api.nullstack.dev/v1/events",
    headers={"Authorization": f"Bearer {KEY}"},
    json={"name": "deploy.start"}
)

            
curl -X POST https://api.nullstack.dev/v1/events \
  -H "Authorization: Bearer $KEY" \
  -d '{"name":"deploy.start"}'

            
          

        


        
        

          

            
Query state

            
Read live resource state with cursor-based pagination built in.

          

          

            
            
            
          

          

            
const { data } = await ns.query("resources", {
  filter: { status: "active" },
  limit: 50
});
// → { data: [...], cursor: "ck_8f2a" }

            
data = ns.query(
    "resources",
    filter={"status": "active"},
    limit=50
)
# → {"data": [...], "cursor": "ck_8f2a"}

            
curl https://api.nullstack.dev/v1/resources \
  -H "Authorization: Bearer $KEY" \
  -G -d "status=active" -d "limit=50"

            
          

        


        
        

          

            
Stream logs

            
Tail production logs over a single long-lived connection.

          

          

            
            
            
          

          

            
const stream = ns.logs.tail({ service: "api" });
for await (const line of stream) {
  console.log(line.message);
}

            
for line in ns.logs.tail(service="api"):
    print(line["message"])

            
curl -N https://api.nullstack.dev/v1/logs/tail?service=api \
  -H "Authorization: Bearer $KEY"
