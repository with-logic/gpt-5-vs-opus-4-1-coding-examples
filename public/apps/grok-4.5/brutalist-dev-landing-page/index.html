PRIM.03

Edge Functions

Deploy JS/WASM to 40 regions. Cold starts under 5 ms. No containers to manage.

Copied! 
// edge/hello.js
export default {
  async fetch(req, env) {
    const { name } = await req.json();
    return Response.json({
      msg: `Hello, ${name}`,
      region: env.REGION
    });
  }
};

// Deploy
// $ brutal deploy ./edge
# edge/hello.py
def handler(req, env):
  name = req.json()["name"]
  return {
    "msg": f"Hello, {name}",
    "region": env.region
  }

# Deploy
# $ brutal deploy ./edge
# Invoke deployed edge fn
curl -X POST \
  https://fn.brutal.dev/hello \
  -H "Content-Type: application/json" \
  -d '{"name":"Ada"}'
