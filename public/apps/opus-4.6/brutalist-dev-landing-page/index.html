from brutal import Auth

  user="dev@brutal.sh",

Deploy

          
Zero-config deployment

import { deploy } from '@brutal/sdk';

const app = await deploy({
  name: 'my-service',
  runtime: 'node:20',
  regions: ['us-east', 'eu-west'],
  scale: { min: 1, max: 100 }
});

// Deployed in 4.2s
// https://my-service.brutal.sh

          
from brutal import Deploy

app = Deploy.create(
  name="my-service",
  runtime="python:3.12",
  regions=["us-east", "eu-west"],
  scale={"min": 1, "max": 100}
)

# Deployed in 4.2s
# https://my-service.brutal.sh

    "name": "my-service",
    "runtime": "node:20",
    "regions": ["us-east","eu-west"],
    "scale": {"min":1,"max":100}
  }'

Observe

          
Real-time metrics stream

        

        

          
          
          
        

        

          

            
            Copied!
          

          
import { observe } from '@brutal/sdk';

const stream = observe.metrics({
  service: 'my-service',
  interval: '1s',
  fields: ['p99', 'errors', 'rps']
});

stream.on('data', m =>
  console.log(`${m.rps} req/s`)
);

          
from brutal import Observe

stream = Observe.metrics(
  service="my-service",
  interval="1s",
  fields=["p99", "errors", "rps"]
)

for m in stream:
    print(f"{m.rps} req/s")

          
curl -N --http2 \
  https://api.brutal.sh/observe \
  -H "Authorization: Bearer $KEY" \
  -d '{
    "service": "my-service",
    "interval": "1s",
    "fields": ["p99","errors","rps"]
  }'

# Stream: {"rps":12840,...}
