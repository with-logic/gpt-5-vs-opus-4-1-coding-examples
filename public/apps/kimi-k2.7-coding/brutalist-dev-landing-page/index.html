from brutalist import Client

api = Client('KEY')
api.build(repo='my-app', tier='scale')

          
curl -X POST https://api.brutalist.dev/v1/build \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $KEY" \

deploy = api.deploy('bx-94821')

print(deploy.url)

          
curl -X POST https://api.brutalist.dev/v1/deploy \
  -H "Authorization: Bearer $KEY" \

  console.log(log.time, log.level);
}

    print(log.time, log.level)

          
curl -N https://api.brutalist.dev/v1/logs \
  -H "Authorization: Bearer $KEY" \
