Features / code first

          

            Three brutally honest primitives: projects, webhooks, and observability—each one shippable from a single snippet.
          

        

        

          

            

              
01 · Projects

              
Spin up an API in one call.

            

            

              
              
              
            

            

              
import { client } from "@brut/dev";

const api = client({ token: process.env.BRUT_KEY });

const project = await api.projects.create({
  name: "landing-page",
  region: "fra1",
});

console.log(project.id);
              

              
from brut import Client
import os

api = Client(token=os.environ["BRUT_KEY"])

project = api.projects.create(
    name="landing-page",
    region="fra1",
)

print(project.id)
              

              
curl https://api.brut.dev/v1/projects \
  -H "Authorization: Bearer $BRUT_KEY" \
  -H "Content-Type: application/json" \
  -d '{"name":"landing-page","region":"fra1"}'

02 · Webhooks

              
Hard-edged webhooks with retries.

            

            

              
              
              
            

            

              
import { client } from "@brut/dev";

const api = client({ token: process.env.BRUT_KEY });

await api.webhooks.subscribe({
  url: "https://grid.app/hook",
  events: ["project.deployed"],
  retry_window: "24h",
});
              

              
from brut import Client

api = Client(token="sk_test_***")

api.webhooks.subscribe(
    url="https://grid.app/hook",
    events=["project.deployed"],
    retry_window="24h",
)
              

              
curl https://api.brut.dev/v1/webhooks \
  -H "Authorization: Bearer $BRUT_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "url":"https://grid.app/hook",
    "events":["project.deployed"],
    "retry_window":"24h"
  }'

03 · Observability

              
Logs that read like blueprints.

            

            

              
              
              
            

            

              
import { client } from "@brut/dev";

const api = client({ token: process.env.BRUT_KEY });

const logs = await api.logs.query({
  project: "landing-page",
  window: "5m",
});

console.table(logs.rows);
              

              
from brut import Client

api = Client(token="sk_live_***")

logs = api.logs.query(
    project="landing-page",
    window="5m",
)

for row in logs.rows:
    print(row["ts"], row["status"], row["latency_ms"])
              

              
curl "https://api.brut.dev/v1/logs?project=landing-page&window=5m" \
  -H "Authorization: Bearer $BRUT_KEY"
