02 / Scale Auto-scale
await api.scale('user-service', { min: 2, max: 100, metrics: { cpu: 70, latency: 200 } }); // Zero-to-N in 400ms
client.scale( "user-service", min=2, max=100, metrics={"cpu": 70, "latency": 200} ) # Zero-to-N in 400ms
curl -X PATCH https://api.brutalist.dev/v1/services/user-service/scale \ -H "Authorization: Bearer key_***" \ -d '{ "min": 2, "max": 100, "metrics": {"cpu": 70, "latency": 200} }'
03 / Monitor Observability
const metrics = await api.metrics({ service: 'user-service', window: '1h', fields: ['p99', 'errors', 'rpm'] }); metrics.alerts.configure({ channel: 'slack', threshold: 'p99 > 500ms' });
metrics = client.metrics( service="user-service", window="1h", fields=["p99", "errors", "rpm"] ) metrics.alerts.configure( channel="slack", threshold="p99 > 500ms" )
curl https://api.brutalist.dev/v1/metrics?service=user-service&window=1h \ -H "Authorization: Bearer key_***" | jq '.p99, .errors, .rpm' curl -X POST https://api.brutalist.dev/v1/alerts \ -d '{"channel":"slack","threshold":"p99 > 500ms"}'
