Authenticate

Scoped tokens, verified in one round trip.

import { Brut } from "@brut/sdk";

export const client = new Brut({
  token: process.env.BRUT_TOKEN,
});

const session = await client.auth.verify({
  scopes: ["deploy", "read"],
});

console.log(session.team);

import os
from brut import Brut

client = Brut(token=os.environ["BRUT_TOKEN"])

session = client.auth.verify(
    scopes=["deploy", "read"],
)

print(session.team)
# "acme-core" · verified in 9ms
curl -s https://api.brut.dev/v1/auth/verify \
  -H "Authorization: Bearer $BRUT_TOKEN" \
  -d '{"scopes":["deploy","read"]}'

# <- 200 OK

Query

SQL over live edge metrics. No pipeline.

const rows = await client.sql`
  SELECT region, p95_ms
  FROM   metrics
  WHERE  service = 'edge-api'
  ORDER  BY p95_ms ASC
  LIMIT  3
`;

rows.forEach((r) => console.log(r));
// { region: "iad", p95_ms: 9 } ...
rows = client.sql("""
    SELECT region, p95_ms FROM metrics
    WHERE service = 'edge-api'
    ORDER BY p95_ms ASC
    LIMIT 3
""")

for row in rows:
    print(row)
# {"region": "iad", "p95_ms": 9} ...
curl -s https://api.brut.dev/v1/sql \
  -H "Authorization: Bearer $BRUT_TOKEN" \
  -d '{"q":"SELECT region,p95_ms FROM metrics LIMIT 3"}'

# <- 200 OK
# [{"region":"iad","p95_ms":9}, ...]
