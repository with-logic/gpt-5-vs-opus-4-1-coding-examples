01 — Deploy REST API
Deploy any function to a global edge network with a single API call. Auto-scaling, zero cold starts.
// Deploy a function
import { deploy } from '@devpwr/sdk';

const app = await deploy({
  name: 'my-api',
  runtime: 'edge',
  regions: ['us-east', 'eu-west'],
  handler: './index.ts'
});

console.log(app.url);
// → https://my-api.devpwr.run
# Deploy a function
from devpwr import deploy

app = deploy(
    name="my-api",
    runtime="edge",
    regions=["us-east", "eu-west"],
    handler="./index.py"
)

print(app.url)
# → https://my-api.devpwr.run
# Deploy a function
curl -X POST https://api.devpwr.com/v1/deploy \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "my-api",
    "runtime": "edge",
    "regions": ["us-east","eu-west"],
    "handler": "./index.ts"
  }'
02 — Query DATABASE
Serverless database with type-safe queries. Branch, preview, and roll back your schema in seconds.
// Type-safe database query
import { db } from '@devpwr/db';

const users = await db
  .select()
  .from('users')
  .where({ active: true })
  .limit(100);

// → [{ id: 1, name: "Ada" }, ...]
# Type-safe database query
from devpwr import db

users = (
    db.select()
    .from_("users")
    .where(active=True)
    .limit(100)
)

# → [{ "id": 1, "name": "Ada" }, ...]
# Query the database
curl https://db.devpwr.com/v1/query \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "table": "users",
    "filter": { "active": true },
    "limit": 100
  }'
03 — Auth AUTH
Drop-in authentication with OAuth, magic links, and passkeys. Session management built in.
// Authenticate a user
import { auth } from '@devpwr/auth';

const session = await auth.signIn({
  email: 'ada@example.com',
  provider: 'magic-link'
});

// → { token: "eyJ...", user: {...} }
# Authenticate a user
from devpwr import auth

session = auth.sign_in(
    email="ada@example.com",
    provider="magic-link"
)

# → { "token": "eyJ...", "user": {...} }
# Create a session
curl -X POST https://auth.devpwr.com/v1/session \
  -H "Content-Type: application/json" \
  -d '{
    "email": "ada@example.com",
    "provider": "magic-link"
  }'
