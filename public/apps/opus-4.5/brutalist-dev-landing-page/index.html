Authentication

Secure API key management with rate limiting built-in.

Copied! 
import { Brutal } from '@brutal/sdk';

const client = new Brutal({
  apiKey: process.env.BRUTAL_KEY
});

const user = await client.auth.verify(token);
Copied! 
from brutal import Brutal

client = Brutal(
    api_key=os.environ["BRUTAL_KEY"]
)

user = client.auth.verify(token)
Copied! 
curl -X POST https://api.brutal.dev/auth \
  -H "Authorization: Bearer $BRUTAL_KEY" \
  -H "Content-Type: application/json" \
  -d '{"token": "user_token_here"}'

Data Fetch

Query any endpoint with automatic caching and retries.

Copied! 
const data = await client.query({
  endpoint: '/users',
  filter: { status: 'active' },
  limit: 100
});

console.log(data.results);
Copied! 
data = client.query(
    endpoint="/users",
    filter={"status": "active"},
    limit=100
)

print(data.results)
Copied! 
curl "https://api.brutal.dev/users?status=active&limit=100" \
  -H "Authorization: Bearer $BRUTAL_KEY"

Webhooks

Real-time event streaming with guaranteed delivery.

Copied! 
await client.webhooks.create({
  url: 'https://your-app.com/hook',
  events: ['user.created', 'order.paid'],
  secret: process.env.WEBHOOK_SECRET
});
Copied! 
client.webhooks.create(
    url="https://your-app.com/hook",
    events=["user.created", "order.paid"],
    secret=os.environ["WEBHOOK_SECRET"]
)
Copied! 
curl -X POST https://api.brutal.dev/webhooks \
  -H "Authorization: Bearer $BRUTAL_KEY" \
  -H "Content-Type: application/json" \
  -d '{"url":"https://your-app.com/hook",
       "events":["user.created","order.paid"]}'
