Ingest

                  
Post one event. Get schema validation, idempotency, and durable storage. Default is strict.

                

                

                  
                  
                  
                  
                  
                    
                    Copied
                  
                

                

                  
// JS — ingest a structured event (strict mode)
await fetch("https://api.bdp.dev/v1/events", {
  method: "POST",
  headers: {
    "Authorization": "Bearer $TOKEN",
    "Content-Type": "application/json",
    "Idempotency-Key": "evt_01HXM9Z7M3N4P"
  },
  body: JSON.stringify({
    type: "deploy.created",
    actor: { id: "usr_2v3", role: "ci" },
    payload: { sha: "9b7c…", env: "prod", region: "iad" }
  })
});

                  
# Python — same event, same contract
import requests, json

r = requests.post(
  "https://api.bdp.dev/v1/events",
  headers={
    "Authorization": "Bearer $TOKEN",
    "Content-Type": "application/json",
    "Idempotency-Key": "evt_01HXM9Z7M3N4P",
  },
  data=json.dumps({
    "type": "deploy.created",
    "actor": {"id": "usr_2v3", "role": "ci"},
    "payload": {"sha": "9b7c…", "env": "prod", "region": "iad"},
  }),
)
print(r.status_code, r.json())

                  
# cURL — smallest surface area
curl -sS https://api.bdp.dev/v1/events \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -H "Idempotency-Key: evt_01HXM9Z7M3N4P" \
  -d '{
    "type":"deploy.created",
    "actor":{"id":"usr_2v3","role":"ci"},
    "payload":{"sha":"9b7c…","env":"prod","region":"iad"}
  }'

                

              


              

                

                  
Deliver

                  
Signed webhooks with retries, replay-by-id, and per-endpoint rate limits. Deterministic payloads.

                

                

                  
                  
                  
                  
                  
                    
                    Copied
                  
                

                

                  
// JS — create a signed webhook endpoint + subscribe
await fetch("https://api.bdp.dev/v1/hooks/endpoints", {
  method: "POST",
  headers: { "Authorization":"Bearer $TOKEN", "Content-Type":"application/json" },
  body: JSON.stringify({
    url: "https://example.com/webhooks/bdp",
    events: ["deploy.created", "deploy.failed"],
    secret: "whsec_•••",
    retry: { max_attempts: 12, backoff: "exp-jitter" }
  })
});

                  
# Python — verify a signature (HMAC SHA-256)
import hmac, hashlib

def verify(secret, raw_body, signature):
  expected = hmac.new(
    secret.encode("utf-8"),
    raw_body,
    hashlib.sha256
  ).hexdigest()
  return hmac.compare_digest(expected, signature)

                  
# cURL — replay a delivery by id (debug-safe)
curl -sS -X POST https://api.bdp.dev/v1/hooks/replay \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{ "delivery_id":"dlv_01HXM9Z7QZQ2", "mode":"dry-run" }'

                

              


              

                

                  
Observe

                  
Tracing, structured logs, and alerts without a vendor-branded dashboard. Export when you’re ready.

                

                

                  
                  
                  
                  
                  
                    
                    Copied
                  
                

                

                  
// JS — attach trace context to any call
const headers = new Headers({
  "Authorization":"Bearer $TOKEN",
  "Content-Type":"application/json",
  "X-Trace-Id":"trc_01HXM9Z7V9KQ",
  "X-Span-Id":"spn_9cc2"
});
await fetch("https://api.bdp.dev/v1/traces/annotate", {
  method:"POST", headers,
  body: JSON.stringify({ level:"info", msg:"deploy start", tags:{ env:"prod" } })
});

                  
# Python — send a structured metric (cheap + fast)
import requests, json
requests.post(
  "https://api.bdp.dev/v1/metrics",
  headers={"Authorization":"Bearer $TOKEN","Content-Type":"application/json"},
  data=json.dumps({
    "name":"webhook.delivery.latency_ms",
    "value": 42.7,
    "tags": {"endpoint":"primary","region":"iad"}
  }),
)

                  
# cURL — query events by type + time (cursor paging)
curl -sS "https://api.bdp.dev/v1/events?type=deploy.created&since=1h&limit=25" \
  -H "Authorization: Bearer $TOKEN" | jq .data[0]
